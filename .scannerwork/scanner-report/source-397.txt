<script>
    function TogglePasswordVisibility(inputId, iconId) {
        var passwordInput = document.getElementById(inputId);
        //alert(passwordInput);
        var visibilityIcon = document.getElementById(iconId);

        if (passwordInput.type === "password") {
            passwordInput.type = "text";
            visibilityIcon.innerText = "visibility_off";
        } else {
            passwordInput.type = "password";
            visibilityIcon.innerText = "visibility";
        }
    }
</script>
    <script>
        function ubah() {
            var update = document.getElementById('update').checked;
            var box = document.getElementById("ubah_password");
            if (update == true) {
                box.style.display = "block";
                document.getElementById("TextBox5").value = "yes";
            } else if (update == false) {
                box.style.display = "none";
                document.getElementById("TextBox5").value = "no";
            }
        }
    </script>
